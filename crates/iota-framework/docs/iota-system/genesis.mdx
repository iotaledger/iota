## Function `create`

This function will be explicitly called once at genesis.
It will create a singleton IotaSystemState object, which contains
all the information we need in the system.


<pre><code>
<b>fun</b> <Link href="genesis#0x3_genesis_create">create</Link>(iota_system_state_id: <Link href="../iota-framework/object#0x2_object_UID">object::UID</Link>, iota_supply: <Link href="../iota-framework/balance#0x2_balance_Balance">balance::Balance</Link>&lt;<Link href="../iota-framework/iota#0x2_iota_IOTA">iota::IOTA</Link>&gt;, genesis_chain_parameters: <Link href="genesis#0x3_genesis_GenesisChainParameters">genesis::GenesisChainParameters</Link>, genesis_validators: <Link href="../move-stdlib/vector#0x1_vector">vector</Link>&lt;<Link href="genesis#0x3_genesis_GenesisValidatorMetadata">genesis::GenesisValidatorMetadata</Link>&gt;, token_distribution_schedule: <Link href="genesis#0x3_genesis_TokenDistributionSchedule">genesis::TokenDistributionSchedule</Link>, timelock_genesis_label: <Link href="../move-stdlib/option#0x1_option_Option">option::Option</Link>&lt;<Link href="../move-stdlib/string#0x1_string_String">string::String</Link>&gt;, system_timelock_cap: <Link href="../iota-framework/timelock#0x2_timelock_SystemTimelockCap">timelock::SystemTimelockCap</Link>, ctx: &<b>mut</b> <Link href="../iota-framework/tx_context#0x2_tx_context_TxContext">tx_context::TxContext</Link>)

<pre><code>
<b>fun</b> <Link href="genesis#0x3_genesis_create">create</Link>(
    iota_system_state_id: UID,
    <b>mut</b> iota_supply: Balance&lt;IOTA&gt;,
    genesis_chain_parameters: <Link href="genesis#0x3_genesis_GenesisChainParameters">GenesisChainParameters</Link>,
    genesis_validators: <Link href="../move-stdlib/vector#0x1_vector">vector</Link>&lt;<Link href="genesis#0x3_genesis_GenesisValidatorMetadata">GenesisValidatorMetadata</Link>&gt;,
    token_distribution_schedule: <Link href="genesis#0x3_genesis_TokenDistributionSchedule">TokenDistributionSchedule</Link>,
    timelock_genesis_label: Option&lt;String&gt;,
    system_timelock_cap: SystemTimelockCap,
    ctx: &<b>mut</b> TxContext,
) \{
    // Ensure this is only called at <Link href="genesis#0x3_genesis">genesis</Link>
    <b>assert</b>!(ctx.epoch() == 0, <Link href="genesis#0x3_genesis_ENotCalledAtGenesis">ENotCalledAtGenesis</Link>);

    <b>let</b> <Link href="genesis#0x3_genesis_TokenDistributionSchedule">TokenDistributionSchedule</Link> \{
        stake_subsidy_fund_nanos,
        allocations,
    } = token_distribution_schedule;

    <b>let</b> subsidy_fund = iota_supply.split(stake_subsidy_fund_nanos);
    <b>let</b> <Link href="storage_fund#0x3_storage_fund">storage_fund</Link> = <Link href="../iota-framework/balance#0x2_balance_zero">balance::zero</Link>();

    // Create all the `Validator` structs
    <b>let</b> <b>mut</b> validators = <Link href="../move-stdlib/vector#0x1_vector">vector</Link>[];
    <b>let</b> count = genesis_validators.length();
    <b>let</b> <b>mut</b> i = 0;
    <b>while</b> (i &lt; count) \{
        <b>let</b> <Link href="genesis#0x3_genesis_GenesisValidatorMetadata">GenesisValidatorMetadata</Link> \{
            name,
            description,
            image_url,
            project_url,
            iota_address,
            gas_price,
            commission_rate,
            protocol_public_key,
            proof_of_possession,
            network_public_key,
            worker_public_key,
            network_address,
            p2p_address,
            primary_address,
            worker_address,
        } = genesis_validators[i];

        <b>let</b> <Link href="validator#0x3_validator">validator</Link> = <Link href="validator#0x3_validator_new">validator::new</Link>(
            iota_address,
            protocol_public_key,
            network_public_key,
            worker_public_key,
            proof_of_possession,
            name,
            description,
            image_url,
            project_url,
            network_address,
            p2p_address,
            primary_address,
            worker_address,
            gas_price,
            commission_rate,
            ctx
        );

        // Ensure that each <Link href="validator#0x3_validator">validator</Link> is unique
        <b>assert</b>!(
            !<Link href="validator_set#0x3_validator_set_is_duplicate_validator">validator_set::is_duplicate_validator</Link>(&validators, &<Link href="validator#0x3_validator">validator</Link>),
            <Link href="genesis#0x3_genesis_EDuplicateValidator">EDuplicateValidator</Link>,
        );

        validators.push_back(<Link href="validator#0x3_validator">validator</Link>);

        i = i + 1;
    };

    // Allocate tokens and staking operations
    <Link href="genesis#0x3_genesis_allocate_tokens">allocate_tokens</Link>(
        iota_supply,
        allocations,
        &<b>mut</b> validators,
        timelock_genesis_label,
        ctx
    );

    // Activate all validators
    <Link href="genesis#0x3_genesis_activate_validators">activate_validators</Link>(&<b>mut</b> validators);

    <b>let</b> system_parameters = <Link href="iota_system_state_inner#0x3_iota_system_state_inner_create_system_parameters">iota_system_state_inner::create_system_parameters</Link>(
        genesis_chain_parameters.epoch_duration_ms,
        genesis_chain_parameters.stake_subsidy_start_epoch,

        // Validator committee parameters
        genesis_chain_parameters.max_validator_count,
        genesis_chain_parameters.min_validator_joining_stake,
        genesis_chain_parameters.validator_low_stake_threshold,
        genesis_chain_parameters.validator_very_low_stake_threshold,
        genesis_chain_parameters.validator_low_stake_grace_period,

        ctx,
    );

    <b>let</b> <Link href="stake_subsidy#0x3_stake_subsidy">stake_subsidy</Link> = <Link href="stake_subsidy#0x3_stake_subsidy_create">stake_subsidy::create</Link>(
        subsidy_fund,
        genesis_chain_parameters.stake_subsidy_initial_distribution_amount,
        genesis_chain_parameters.stake_subsidy_period_length,
        genesis_chain_parameters.stake_subsidy_decrease_rate,
        ctx,
    );

    <Link href="iota_system#0x3_iota_system_create">iota_system::create</Link>(
        iota_system_state_id,
        validators,
        <Link href="storage_fund#0x3_storage_fund">storage_fund</Link>,
        genesis_chain_parameters.protocol_version,
        genesis_chain_parameters.chain_start_timestamp_ms,
        system_parameters,
        <Link href="stake_subsidy#0x3_stake_subsidy">stake_subsidy</Link>,
        system_timelock_cap,
        ctx,
    );
}

## Function `allocate_tokens`

</code></pre>



<details>
<summary>Implementation</summary>


<pre><code>
<b>fun</b> <Link href="genesis#0x3_genesis_allocate_tokens">allocate_tokens</Link>(
    <b>mut</b> iota_supply: Balance&lt;IOTA&gt;,
    <b>mut</b> allocations: <Link href="../move-stdlib/vector#0x1_vector">vector</Link>&lt;<Link href="genesis#0x3_genesis_TokenAllocation">TokenAllocation</Link>&gt;,
    validators: &<b>mut</b> <Link href="../move-stdlib/vector#0x1_vector">vector</Link>&lt;Validator&gt;,
    timelock_genesis_label: Option&lt;String&gt;,
    ctx: &<b>mut</b> TxContext,
) \{

    <b>while</b> (!allocations.is_empty()) \{
        <b>let</b> <Link href="genesis#0x3_genesis_TokenAllocation">TokenAllocation</Link> \{
            recipient_address,
            amount_nanos,
            staked_with_validator,
            staked_with_timelock_expiration,
        } = allocations.pop_back();

        <b>let</b> allocation_balance = iota_supply.split(amount_nanos);

        <b>if</b> (staked_with_validator.is_some()) \{
            <b>let</b> validator_address = staked_with_validator.destroy_some();
            <b>let</b> <Link href="validator#0x3_validator">validator</Link> = <Link href="validator_set#0x3_validator_set_get_validator_mut">validator_set::get_validator_mut</Link>(
                validators, validator_address
            );
            <b>if</b> (staked_with_timelock_expiration.is_some()) \{
                <b>let</b> timelock_expiration = staked_with_timelock_expiration.destroy_some();
                <Link href="timelocked_staking#0x3_timelocked_staking_request_add_stake_at_genesis">timelocked_staking::request_add_stake_at_genesis</Link>(
                    <Link href="validator#0x3_validator">validator</Link>,
                    allocation_balance,
                    recipient_address,
                    timelock_expiration,
                    timelock_genesis_label,
                    ctx
                );
            } <b>else</b> \{
                <Link href="validator#0x3_validator">validator</Link>.request_add_stake_at_genesis(
                    allocation_balance,
                    recipient_address,
                    ctx
                );
            }
        } <b>else</b> \{
            <Link href="../iota-framework/iota#0x2_iota_transfer">iota::transfer</Link>(
                allocation_balance.into_coin(ctx),
                recipient_address,
            );
        };
    };
    allocations.destroy_empty();

    // Provided allocations must fully allocate the iota_supply and there
    // should be none left at this point.
    iota_supply.destroy_zero();
}
