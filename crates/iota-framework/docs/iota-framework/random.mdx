<details>
<summary>Implementation</summary>


<pre>
<code>
<b>fun</b> <Link to="../iota-framework/random#0x2_random_update_randomness_state">update_randomness_state</Link>(
    self: &<b>mut</b> <Link to="../iota-framework/random#0x2_random_Random">Random</Link>,
    new_round: u64,
    new_bytes: <Link to="../move-stdlib/vector#0x1_vector">vector</Link>&lt;u8&gt;,
    ctx: &TxContext,
) \{
    // Validator will make a special system call <b>with</b> sender set <b>as</b> 0x0.
    <b>assert</b>!(ctx.sender() == @0x0, <Link to="../iota-framework/random#0x2_random_ENotSystemAddress">ENotSystemAddress</Link>);

    // Randomness should only be incremented.
    <b>let</b> epoch = ctx.epoch();
    <b>let</b> inner = self.<Link to="../iota-framework/random#0x2_random_load_inner_mut">load_inner_mut</Link>();
    <b>if</b> (inner.randomness_round == 0 && inner.epoch == 0 && inner.random_bytes.is_empty()) \{
        // First <b>update</b> should be for round zero.
        <b>assert</b>!(new_round == 0, <Link to="../iota-framework/random#0x2_random_EInvalidRandomnessUpdate">EInvalidRandomnessUpdate</Link>);
    } <b>else</b> \{
        // Subsequent updates should either increase epoch or increment randomness_round.
        // Note that epoch may increase by more than 1 <b>if</b> an epoch is completed without
        // randomness ever being generated in that epoch.
        <b>assert</b>!(
            (epoch &gt; inner.epoch && new_round == 0) ||
                (new_round == inner.randomness_round + 1),
            <Link to="../iota-framework/random#0x2_random_EInvalidRandomnessUpdate">EInvalidRandomnessUpdate</Link>
        );
    };

    inner.epoch = ctx.epoch();
    inner.randomness_round = new_round;
    inner.random_bytes = new_bytes;
}
</code>

    <b>min</b> + (rand % range_size <b>as</b> u128)
}
